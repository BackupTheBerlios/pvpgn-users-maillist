Las reglas de Iptables son:

iptables -A INPUT -s 0.0.0.0/0 -p tcp -m tcp --dport 6112:6119 -j ACCEPT
iptables -A OUTPUT -p tcp -m state --state RELATED,ESTABLISHED -j ACCEPT
iptables -A INPUT -s 0.0.0.0/0 -p udp -m udp --dport 6112:6119 -j ACCEPT
iptables -A OUTPUT -p udp -m state --state RELATED,ESTABLISHED -j ACCEPT

iptables -A INPUT -s 0.0.0.0/0 -p tcp -m tcp --dport 23 -j ACCEPT
iptables -A OUTPUT -p tcp -m state --state RELATED,ESTABLISHED -j ACCEPT
iptables -A INPUT -s 0.0.0.0/0 -p udp -m udp --dport 23 -j ACCEPT
iptables -A OUTPUT -p udp -m state --state RELATED,ESTABLISHED -j ACCEPT

iptables -A INPUT -s 0.0.0.0/0 -p tcp -m tcp --dport 6667 -j ACCEPT
iptables -A OUTPUT -p tcp -m state --state RELATED,ESTABLISHED -j ACCEPT
iptables -A INPUT -s 0.0.0.0/0 -p udp -m udp --dport 6667 -j ACCEPT
iptables -A OUTPUT -p udp -m state --state RELATED,ESTABLISHED -j ACCEPT
