If on Windows, paste the text below into a xxx.bat file and run from  
your "files" directory
..\bnftp --client=SEXP --arch=IX86 --file=SEXP_IX86_1xx_115.mpq  
useast.battle.net 6112
..\bnftp --client=STAR --arch=IX86 --file=STAR_IX86_1xx_115.mpq  
useast.battle.net 6112
..\bnftp --client=SEXP --arch=XMAC --file=SEXP_XMAC_1xx_115.mpq  
useast.battle.net 6112
..\bnftp --client=STAR --arch=XMAC --file=STAR_XMAC_1xx_115.mpq  
useast.battle.net 6112
..\bnftp --client=SEXP --arch=PMAC --file=SEXP_PMAC_1xx_115.mpq  
useast.battle.net 6112
..\bnftp --client=STAR --arch=PMAC --file=STAR_PMAC_1xx_115.mpq  
useast.battle.net 6112


Add these entries to your versioncheck.conf file

# New 16 May 2007
# Starcraft Broodwar (Expansion) 1.14 - (Broodwar mode) - WINDOWS
"A=2521522835 B=3428392135 C=218673704 4 A=A^S B=B-C C=C+A A=A-B" \
"IX86ver1.mpq" \
IX86 \
SEXP \
"StarCraft.exe 05/08/07 19:13:26 1220608" \
0x000000d1 \
0x010f0001 \
0xc9ca8bc3 \
SEXP_115


# New 16 May 2007
# StarCraft - Broodwar (Expansion) 1.15 - (Broodwar Mode) MacOS X
"A=2521522835 B=3428392135 C=218673704 4 A=A^S B=B-C C=C+A A=A-B" \
"XMACver1.mpq" \
XMAC \
SEXP \
"Starcraft (Carbon) 05/08/107 15:40:00 1400487" \
0x000000d1 \
0x010f0001 \
0xe02b5360 \
SEXP_115


Change your autoupdate.conf file as you need.
